Note that the above title becomes the toctree caption, while the filename is what index.rst looks for. Also note that if the output is cleared before pushing the notebook to the repository, the pipeline will execute the notebook and generate the output during the documentation build process. This a good way to ensure the notebooks are up-to-date with the source code.

import numpy as np
import matplotlib.pyplot as plt
from statmechcrack import CrackIsometric
v = np.linspace(1, 11, 33)
_ = plt.figure()
for N in [5, 10, 25, 100]:
    model = CrackIsometric(N=N)
    beta_A_0 = model.beta_A_0_isometric(v, [1, 1])
    _ = plt.plot(v - 1, N**3/3/model.kappa*beta_A_0, label='$N=$'+str(N))
_ = plt.plot(v - 1, (v - 1)**2/2, 'k--', label='limit')
_ = plt.xlabel(r'$\Delta v$')
_ = plt.ylabel(r'$(N^3/3\kappa)\beta\Delta A_0$')
_ = plt.legend()
plt.show()
